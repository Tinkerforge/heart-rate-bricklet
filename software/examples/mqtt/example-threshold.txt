# Change XYZ to the UID of your Heart Rate Bricklet

setup:
	# Get threshold callbacks with a debounce time of 10 seconds (10000ms)
	publish '{"debounce": 10000}' to tinkerforge/request/heart-rate_bricklet/XYZ/set_debounce_period
	
	# Handle incoming heart rate reached callbacks
	subscribe to tinkerforge/callback/heart_rate_bricklet/XYZ/heart_rate_reached
	publish '{"register": true}' to tinkerforge/register/heart_rate_bricklet/XYZ/heart_rate_reached # Register heart_rate_reached callback
	
	# Configure threshold for heart rate "greater than 100 bpm"
	publish '{"option": "greater", "min": 100, "max": 0}' to tinkerforge/request/heart_rate_bricklet/XYZ/set_heart_rate_callback_threshold
